```javascript
function getLatestTradeDate() {
  const today = new Date();
  const date = new Date(today);
  
  // 周六 (6) → 回退 1 天到周五
  if (today.getDay() === 6) {
    date.setDate(date.getDate() - 1);
  }
  // 周日 (0) → 回退 2 天到周五
  else if (today.getDay() === 0) {
    date.setDate(date.getDate() - 2);
  }
  
  return formatDate(date); // 格式：2026-04-10
}
```
