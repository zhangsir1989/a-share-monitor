### 修复 2：优化搜索建议逻辑

**修改前**:
```javascript
// 只返回 10 条
.slice(0, 10);
```

**修改后**:
```javascript
// 返回最多 20 条
// 按匹配度排序：完全匹配优先
results.sort((a, b) => {
  const aExact = a.name === query || a.code === query;
  const bExact = b.name === query || b.code === query;
  if (aExact && !bExact) return -1;
  if (!aExact && bExact) return 1;
  return a.code.localeCompare(b.code);
});

res.json({
  data: results.slice(0, 20),  // 最多 20 条
  count: results.length,        // 总匹配数
  total: results.length
});
```
